14struct Route {
15 std::string source;
16 std::string target;
17 std::vector<std::string> airlines;
18
19 bool operator<(const Route& r) const {
20 if (source != r.source)
21 return source < r.source;
22 if (target != r.target)
23 return target < r.target;
24 return airlines < r.airlines;
25 }
26 bool operator==(const Route& r) const {
27 return source == r.source && target == r.target && airlines == r.airlines;
28 }
29};
